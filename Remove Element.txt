class Solution {
public:
    int removeElement(int A[], int n, int elem) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        
        int shift = 0;
        int idx = 0;
        
        while(idx < n){
            if(A[idx] == elem){
                idx++;
                shift ++;
                continue;
            }
            
            A[idx-shift] = A[idx++];
        }
        
        return idx -shift;
        
    }
};
