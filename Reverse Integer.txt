class Solution {
public:
    int reverse(int x) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        
        int val = abs(x);
        int rev = 0;
        
        
        while(val > 0){
            rev = rev*10 + val%10;
            val /= 10;
        }
        
        return (x>0)?rev:-rev;
    }
};
